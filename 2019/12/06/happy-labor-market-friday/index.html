Happy Labor Market Friday!

December 6, 2019

Here are data from the latest Employment Situation. These graphs attempt to replicate those published by the NY Times.

Unemployment rate or number of unemployed as a percentage of the labor force:

# Load packages
library(tidyquant)
library(ggplot2)
library(ggthemes)

mygray <- "gray66"

ur <- tq_get("UNRATE", get = "economic.data", from = "2010-01-01")
ur <- ur %>% 
  mutate(series = "Unemployment rate")
last_date <- max(ur$date)
last_lvl <- ur$price[[which(ur$date == max(ur$date))]]
ggplot(data = ur) +
  geom_line(mapping = aes(x = date, y = price), size = 1.1, color = mygray) + 
  # Highlight last observation
  geom_point(data = filter(ur, date == last_date), 
             mapping = aes(x = date, y = price),
             size = 5) +
  xlab("") + ylab("Percent") + ggtitle("Unemployment rate") +
  # Label y-axis for wsj theme
  scale_y_continuous(breaks = c(4, 6, 8, 10),
                     labels = c("4", "6", "8", "10%")) +
  # Label last observation 
  annotate("text", x = last_date, y = last_lvl, label = paste0(last_lvl), 
           color = "black", vjust = -1.5) +
  # Use sans serif font instead of mono, use gray color
  theme_wsj(base_family = "sans", color = "gray", title_family = "sans", base_size = 12)

The unemployment rate was last_lvl.

payems <- tq_get("PAYEMS", get = "economic.data", from = "2016-11-01") 
payems <- payems %>% 
  mutate(change_jobs = price - lag(price),
         mycat = date == max(date)) %>% 
  filter(!is.na(change_jobs))
  
last_date <- max(payems$date)
last_lvl <- payems$change_jobs[[which(payems$date == max(payems$date))]]

segment_height <- 100
ggplot(data = payems) +
  geom_col(mapping = aes(x = date, y = change_jobs, color = mycat, fill = mycat),
           inherit.aes = FALSE) +
  scale_y_continuous(breaks = c(100, 200, 300, 400),
                     labels = c("100", "200", "300", "400 thousand"),
                     limits = c(0, 400)) +
  # Label last observation 
  annotate("text", x = last_date, y = last_lvl + segment_height, 
           label = paste0("+", last_lvl, "\nNov 2019"), 
           color = "black") +  
  annotate("segment", x = last_date, xend = last_date, 
           y = last_lvl, yend = last_lvl + segment_height - 25,
           linetype = "dashed") +
  # Color the bars
  scale_fill_manual(values = c(mygray, "black")) +
  scale_color_manual(values = c(mygray, "black")) +
  # Turn off legend
  guides(color = FALSE, fill = FALSE) +
  ggtitle("Monthly change in jobs") +
  theme_wsj(base_family = "sans", color = "gray", title_family = "sans", base_size = 12) +
  # Put grid lines on top
  theme(panel.background = element_rect(fill = NA),
        panel.ontop = TRUE)
