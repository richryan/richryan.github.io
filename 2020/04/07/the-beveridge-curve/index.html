The Beveridge Curve

April 7, 2020

Here are the latest data that depict the negative relationship between unemployment and job vacancies. The scatter plot of unemployment rates versus vacancy rates is known as the Beveridge curve.

# Load packages
library(tidyverse)
library(tidyquant)
library(ggplot2)
library(ggthemes)
library(viridis)

mygray <- "gray66"

# Data series
my_series <- c("JTSJOL", "PAYEMS", "UNRATE")
dat <- tq_get(my_series, get = "economic.data", from = "1900-01-01")
print(unique(dat$symbol))

## [1] "JTSJOL" "PAYEMS" "UNRATE"

# Data for recession shading
recess <- tq_get("USRECM", get = "economic.data", from = "1800-01-01")
tail(recess)

## # A tibble: 6 x 3
##   symbol date       price
##   <chr>  <date>     <int>
## 1 USRECM 2019-10-01     0
## 2 USRECM 2019-11-01     0
## 3 USRECM 2019-12-01     0
## 4 USRECM 2020-01-01     0
## 5 USRECM 2020-02-01     0
## 6 USRECM 2020-03-01     0

recess_dat <- recess %>% 
  arrange(date) %>% 
  mutate(same = 1 - (price == lag(price))) %>% 
  # Remove first row, an NA, for cumulative sum
  filter(date > min(recess$date)) %>% 
  mutate(era = cumsum(same)) %>% 
  # Filter only recessions
  filter(price == 1)

recess_dat <- recess_dat %>% 
  group_by(era) %>% 
  # Unncessary, but to be sure...
  arrange(date) %>% 
  filter(row_number() == 1 | row_number() == n())

# Now reshape the data wide.
# Each row will contain the start and end dates of a recession.
recess_dat <- recess_dat %>% 
  mutate(junk = row_number()) %>% 
  mutate(begin_end = case_when(
    junk == 1 ~ "begin",
    junk == 2 ~ "end"
  ))

recess_wide <- recess_dat %>%
  ungroup() %>% 
  select(symbol, price, date, era, begin_end) %>% 
  pivot_wider(names_from = begin_end, values_from = date)

Here is the unemployment-rate series:

dat_ur <- filter(dat, symbol == "UNRATE")
ggplot(data = dat_ur) +
  geom_line(mapping = aes(x = date, y = price)) +
  geom_rect(data = filter(recess_wide,
                          begin >= min(dat_ur$date)), 
            mapping = aes(xmin = begin, xmax = end,
                          ymin = -Inf, ymax = Inf),
            fill = "blue", alpha = 0.2) +
  xlab("") + ylab("Percent") + ggtitle("Unemployment rate")

Here is the vacancy-rate series:

dat_wide <- dat %>% 
  pivot_wider(names_from = symbol, values_from = price) %>% 
  mutate(vac = 100*JTSJOL / (JTSJOL + PAYEMS)) %>% 
  drop_na(vac)
tail(dat_wide)

ggplot(data = dat_wide) +
  geom_line(mapping = aes(x = date, y = vac)) +
  geom_rect(data = filter(recess_wide,
                          begin >= min(dat_wide$date)), 
            mapping = aes(xmin = begin, xmax = end,
                          ymin = -Inf, ymax = Inf),
            fill = "blue", alpha = 0.2) +
  xlab("") + ylab("Percent") + ggtitle("Vacancy rate")

Here’s the Beveridge curve:

# Subset based on last recession
latest_recess <- max(recess_wide$begin)
dat_wide <- dat_wide %>% 
  mutate(late = case_when(
    date >= latest_recess ~ paste(month(latest_recess, label = TRUE),
                                  year(latest_recess),
                                  "on"),
    TRUE ~ "Early period")) 

ggplot(dat_wide) +
  geom_path(mapping = aes(x = UNRATE, y = vac), color = "gray") + 
  geom_point(mapping = aes(x = UNRATE, y = vac, color = late)) +
  guides(color = FALSE) +
  xlab("Unemployment rate") + ylab("Job vacancy rate") + 
  ggtitle("Beveridge curve") +
  scale_color_viridis(discrete=TRUE) +
  theme_bw()
