Facts from the Employment Situation

April 8, 2020

Here are a few facts from the most recent Employment Situation. The latest data on labor-force participation, highlighted in this Brookings post, and the employment-population ratio are depicted below.

# Load packages
library(tidyverse)
library(tidyquant)
library(ggplot2)

# Data for recession shading
recess <- tq_get("USRECM", get = "economic.data", from = "1800-01-01")
tail(recess)

## # A tibble: 6 x 3
##   symbol date       price
##   <chr>  <date>     <int>
## 1 USRECM 2019-10-01     0
## 2 USRECM 2019-11-01     0
## 3 USRECM 2019-12-01     0
## 4 USRECM 2020-01-01     0
## 5 USRECM 2020-02-01     0
## 6 USRECM 2020-03-01     0

recess_dat <- recess %>% 
  arrange(date) %>% 
  mutate(same = 1 - (price == lag(price))) %>% 
  # Remove first row, an NA, for cumulative sum
  filter(date > min(recess$date)) %>% 
  mutate(era = cumsum(same)) %>% 
  # Filter only recessions
  filter(price == 1)

recess_dat <- recess_dat %>% 
  group_by(era) %>% 
  # Unncessary, but to be sure...
  arrange(date) %>% 
  filter(row_number() == 1 | row_number() == n())

# Now reshape the data wide.
# Each row will contain the start and end dates of a recession.
recess_dat <- recess_dat %>% 
  mutate(junk = row_number()) %>% 
  mutate(begin_end = case_when(
    junk == 1 ~ "begin",
    junk == 2 ~ "end"
  ))

recess_wide <- recess_dat %>%
  ungroup() %>% 
  select(symbol, price, date, era, begin_end) %>% 
  pivot_wider(names_from = begin_end, values_from = date)

dat <- tq_get(c("CIVPART", "EMRATIO"), 
              get = "economic.data", from = "1900-01-01")

# Name series
dat <- dat %>% 
  mutate(Series = case_when(
    symbol == "CIVPART" ~ "Labor-force participation rate",
    symbol == "EMRATIO" ~ "Employment-population ratio"
  ))

dat2plt <- dat %>% 
  mutate(last = date == max(date))

ggplot() +
  geom_path(data = dat2plt, 
            mapping = aes(x = date, y = price, color = Series),
            key_glyph = "timeseries") +
  geom_point(filter(dat2plt, last == 1),
             mapping = aes(x = date, y = price)) +
  labs(x = "", y = "Percent") +
  geom_rect(data = filter(recess_wide,
                          begin >= min(dat2plt$date)),
            mapping = aes(xmin = begin, xmax = end,
                          ymin = -Inf, ymax = Inf),
            fill = "blue", alpha = 0.2) +
  theme_bw()

GREG MANKIW’S BLOG: Random Observations for Students of Economics compares the number of job losers on temporary layoff to the number of permanent job losers. To facilitate the comparison between the two job-loss series, I indexed the previous month to 100 for both series. (I stole the title of this post from the post.)

The job-loss series are depicted below. To make the graph, it is important to know that both series are available at a monthly frequency, seasonally adjusted, and quoted in thousands of persons.

# Grab the 2nd-to-last date, uneloquently
last2 <- nth(loss$date, -2L)
dat_loss <- loss %>% 
  group_by(symbol) %>% 
  mutate(index_maker = max(price*(date == last2)),
         index = 100*price/index_maker) %>% 
  mutate(`Unemployment level` = case_when(
    symbol == "LNS13023653" ~ "Job losers on layoff",
    symbol == "LNS13026638" ~ "Permanent job losers"
  ))

ggplot(filter(dat_loss, date >= ymd("2015-01-01"))) +
  geom_line(mapping = aes(x = date, y = index, 
                          color = `Unemployment level`),
            key_glyph = "timeseries") +
  labs(x = "", 
       y = paste0("Index (", 
                  month(last2, label = TRUE), 
                  " ", year(last2), 
                  "= 100)")) +
  theme_bw()
